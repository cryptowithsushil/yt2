            const languageSelect = document.getElementById('language');

            };
            
            getTranscriptBtn.addEventListener('click', function() {
                const url = document.getElementById('videoUrl').value;
                const language = languageSelect.value;
                const languageNames = { 'en': 'English', 'hi': 'Hindi', 'es': 'Spanish', 'fr': 'French' };
                
                if (!url) {
                    transcriptText.innerHTML = '<p style="color: red; text-align: center;">Please enter a YouTube URL</p>';
                    return;
                }
                
                loadingLanguage.textContent = languageNames[language];
                loadingDiv.style.display = 'block';
                transcriptText.innerHTML = '';
                
                // Simulate API call
                setTimeout(() => {
                    loadingDiv.style.display = 'none';
                    transcriptText.innerHTML = `
                        <h3>Transcript (${languageNames[language]})</h3>
                        <p>${mockTranscripts[language] || mockTranscripts['en']}</p>
                        <p style="margin-top: 20px; font-style: italic; color: #666;">
                            Note: This is a demo transcript. In a production environment, 
                            this would connect to a backend service that extracts actual YouTube transcripts.
                        </p>
                    `;
                }, 2000);
            });
            
            copyBtn.addEventListener('click', function() {
                const textToCopy = transcriptText.innerText;
                navigator.clipboard.writeText(textToCopy).then(() => {
                    const originalText = copyBtn.innerHTML;
                    copyBtn.innerHTML = '<i class="fas fa-check"></i> Copied!';
                    setTimeout(() => {
                        copyBtn.innerHTML = originalText;
                    }, 2000);
                });
            });
            
            downloadBtn.addEventListener('click', function() {
                const textToDownload = transcriptText.innerText;
                const blob = new Blob([textToDownload], { type: 'text/plain' });
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = 'youtube_transcript.txt';
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                URL.revokeObjectURL(url);
            });
        });
